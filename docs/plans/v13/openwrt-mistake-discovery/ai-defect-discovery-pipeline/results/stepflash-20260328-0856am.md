```c
#include <stdlib.h>
#include <string.h>
#include <libubus.h>
#include <blobmsg.h>

struct iface_state {
    char *name;    /* interface name */
    char *mac;     /* MAC address */
    char *ipaddr;  /* IPv4 address */
};

static const struct blobmsg_policy iface_state_policy[] = {
    { .name = "name",   .type = BLOBMSG_STRING },
    { .name = "mac",    .type = BLOBMSG_STRING },
    { .name = "ipaddr", .type = BLOBMSG_STRING },
};

static int iface_state_parse(struct blob_attr *msg, struct iface_state *s)
{
    struct blob_attr *tb[ARRAY_SIZE(iface_state_policy)];
    blobmsg_parse(iface_state_policy, ARRAY_SIZE(iface_state_policy), tb, msg, 0);

    if (!tb[0]) return -1;  /* name is required */

    s->name = strdup(blobmsg_get_string(tb[0]));
    if (tb[1]) s->mac = strdup(blobmsg_get_string(tb[1]));
    if (tb[2]) s->ipaddr = strdup(blobmsg_get_string(tb[2]));

    return s->name ? 0 : -1;
}

struct iface_state *iface_state_alloc(struct blob_attr *msg)
{
    struct iface_state *s = calloc(1, sizeof(*s));
    if (!s) return NULL;
    if (iface_state_parse(msg, s) < 0) {
        iface_state_free(s);
        return NULL;
    }
    return s;
}

void iface_state_free(struct iface_state *s)
{
    if (!s) return;
    free(s->name);
    free(s->mac);
    free(s->ipaddr);
    free(s);
}
```

**Explanation**: This allocates a new network interface state structure from a blob message (nested dictionary) received via ubus. It uses `blobmsg_parse` with a policy to extract strings, duplicates them into the struct, and returns the populated object. If parsing fails or memory allocation fails, it cleans up and returns `NULL`. The caller must later call `iface_state_free` to release memory.
- The `blobmsg_policy` array defines expected keys and their types. The parse function sets fields only if the corresponding attribute is present.
- Challenges: Memory management: every `strdup` can fail; we handle that in the allocation wrapper. Also, `ARRAY_SIZE` macro must be defined (from `linux/kernel.h` or custom); we implicitly assume it.
- Debate: Should the structure contain more fields like `mtu` or `type`? The snippet covers the basics.
